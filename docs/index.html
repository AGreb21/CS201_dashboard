```{r}
griffeydata <- read_csv(here("data/KenGriffyJr.csv"))
griffeydata <- griffeydata %>%
  clean_names()
```

```{r}
griffeydata <- griffeydata[-c(21,22), ]
griffeydata <- griffeydata %>%
  mutate(tm = ifelse(tm == "TOT", "CIN", tm)) %>%
  mutate(lg = ifelse(lg == "MLB", "NL", lg))
```
